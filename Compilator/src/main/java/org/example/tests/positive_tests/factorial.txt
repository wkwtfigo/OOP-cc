class Factorial is
    method fact(n: Integer) : Integer is
        if n.LessEqual(Integer(1)) then
            return Integer(1)
        else
            return n.Mult(fact(n.Minus(Integer(1))))
        end
    end
end

class Main is
    method start() is
        var f : Factorial()
        var res : Integer = f.fact(Integer(5))
        print res  // Должно вывести 120
    end
end
