class Base is
    method whoAmI() : Integer => Integer(1)
end

class Main extends Base is
    method whoAmI() : Integer => Integer(7)
    var a : Integer()

    this() is
        a := Integer(0)
    end

    this(k : Integer) is
        a := k
    end 

    method start() is
        var x : Base()
        var z : Integer(45)
        var y : Main(z)
        print y.whoAmI()
        var res : x.whoAmI()
        print res
    end
end
