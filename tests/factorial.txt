class Factorial is
    method fact(n: Integer) : Integer is
        if n.LessEqual(Integer(1)) then
            return Integer(1)
        else
            return n.Mult(fact(n.Minus(Integer(1))))
        end
    end
end

class Main is
    this() is
        var f : Factorial()
        var res : f.fact(Integer(5))   // 120
    end
end
